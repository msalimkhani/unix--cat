#include <stdio.h>

int readfile(const char *__buf)
{
    FILE *ptr;
    char ch;
    ptr = fopen(__buf,"r");
    if(NULL == ptr)
    {
        printf("Error reading file \"%s\"\n",__buf);
        return 127;
    }
    do
    {
        ch = fgetc(ptr);
        if(ch==EOF)
            break;
        printf("%c",ch);
    } while (ch != EOF);
    fclose(ptr);
    return 0;
}

int main(int argc, char *argv[])
{
    if(argc==2)
    {
        return readfile(argv[1]);
    }
    else
    {
        return 1;
    }
}
